% Macro for drawing 3D Convolution Blocks
% Usage: \convblock{x}{y}{width}{height}{depth}
\def\convblock#1#2#3#4#5{
    \pgfmathsetmacro{\w}{#3/2}
    \pgfmathsetmacro{\h}{#4/2}
    \pgfmathsetmacro{\dx}{#5*0.6}
    \pgfmathsetmacro{\dy}{#5*0.4}
    
    % Top face
    \filldraw[fill=orange!50!brown, draw=brown!80!black, thick, line join=round]
        (#1-\w, #2+\h) -- (#1+\w, #2+\h) -- (#1+\w+\dx, #2+\h+\dy) -- (#1-\w+\dx, #2+\h+\dy) -- cycle;
    % Right face
    \filldraw[fill=orange!70!brown, draw=brown!80!black, thick, line join=round]
        (#1+\w, #2-\h) -- (#1+\w, #2+\h) -- (#1+\w+\dx, #2+\h+\dy) -- (#1+\w+\dx, #2-\h+\dy) -- cycle;
    % Front face
    \filldraw[fill=orange!40!brown, draw=brown!80!black, thick, line join=round]
        (#1-\w, #2-\h) rectangle (#1+\w, #2+\h);
}
